    ```typescript
    function normalizeCategories(raw: unknown[]): string[] {
      return raw.flatMap((c) => {
        if (typeof c === 'string') return [c]
        if (typeof c === 'object' && c !== null) {
          return Object.values(c as Record<string, string>).filter(Boolean)
        }
        return []
      })
    }
    ```
